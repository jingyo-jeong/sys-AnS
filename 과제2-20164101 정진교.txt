//1 밑원의 반지름과 높이를 입력 받아 원기둥의 부피를 구하시오.
//(단, 매크로 VOLUME(x,y)은 반지름이 x이고 높이 y인 원기둥의 부피

#include <stdio.h>
#define _CRT_SECURE_NO_WARNINGS
#define VOLUME(x,y)	(3.14*(x)*(x)*(y))

void main() {
	int r, h;
	double bupi;

	printf("밑원의 반지를괌 높이를 입력하시오 \n");
	scanf_s("%d %d" , &r , &h);
	bupi = VOLUME(r, h);

	printf("부피 : %lf" , bupi);

}

void main() {
	int a, b, c , max;
	printf("세 수를 입력하세요 \n");
	scanf_s("%d %d %d", &a, &b, &c);

// 몸무게와 키를 입력 받아 bmi를 계산하시오 
// BMI = 몸구게 / 키 제곱 , 몸무게 :KG 키:M 단위 
#include <stdio.h>
#include <math.h>

void main() {
	double weight, height ,res;

	printf("몸무게와 키를 입력 하세요 \n");
	scanf_s("%lf %lf", &weight ,&height);

	height = height / 100; // m로 계산

	res = weight /pow(height,2);
	// res = weight / (height * height);

	printf("Bmi 지수는 : %lf 입니다", res);


}

// 구의 부피와 표면적을 구하는데 구의 반지름을 표준입력으로 처리하시오.

#define PI 3.14

void main() {
	int r, v, s;

	printf("반지름을 입력하세요 :");
	scanf_s("%d", &r);
